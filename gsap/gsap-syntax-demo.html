GSAP 语法汇总与演示

1. 基本动画方法

TO
FROM
FROMTO
// gsap.to() - 从当前状态动画到目标状态 gsap.to("#box1", { x: 200, y: 50, rotation: 360, duration: 2, ease: "bounce.out" }); // gsap.from() - 从指定状态动画到当前状态 gsap.from("#box2", { x: -200, y: -50, scale: 0, duration: 2, ease: "elastic.out" }); // gsap.fromTo() - 从指定状态动画到另一个指定状态 gsap.fromTo("#box3", { x: -100, scale: 0, rotation: -180 }, { x: 100, scale: 1.5, rotation: 180, duration: 2, ease: "power2.inOut" } );

2. 时间线 (Timeline)

1
2
3
4
// 创建时间线 const tl = gsap.timeline({ repeat: -1, yoyo: true, onUpdate: () => { // 更新进度条 const progress = tl.progress() * 100; document.getElementById('timelineBar').style.width = progress + '%'; } }); // 添加动画到时间线 tl.to("#circle1", { y: -100, duration: 0.5 }) .to("#circle2", { y: -100, duration: 0.5 }, "-=0.3") // 重叠时间 .to("#circle3", { y: -100, duration: 0.5 }, "-=0.3") .to("#circle4", { y: -100, duration: 0.5 }, "-=0.3"); // 时间线控制 tl.pause(); // 暂停 tl.resume(); // 继续 tl.restart(); // 重新开始

3. 缓动函数 (Easing)

EASE
BOUNCE
ELASTIC
BACK
// 不同的缓动函数 gsap.to("#ease1", { x: 300, duration: 2, ease: "power2.inOut" }); gsap.to("#ease2", { x: 300, duration: 2, ease: "bounce.out" }); gsap.to("#ease3", { x: 300, duration: 2, ease: "elastic.out(1, 0.3)" }); gsap.to("#ease4", { x: 300, duration: 2, ease: "back.out(1.7)" }); // 常用缓动函数: // "none" - 线性 // "power1.out", "power2.out", "power3.out", "power4.out" // "bounce.out", "elastic.out", "back.out" // "circ.out", "expo.out", "sine.out"

4. 变换动画 (Transform)

旋转
缩放
倾斜
// 旋转 gsap.to("#transform1", { rotation: 360, duration: 2, ease: "power2.inOut" }); // 缩放 gsap.to("#transform2", { scale: 1.5, duration: 2, ease: "elastic.out", yoyo: true, repeat: 1 }); // 倾斜 gsap.to("#transform3", { skewX: 30, skewY: 10, duration: 2, ease: "back.out" });

5. 颜色和文本动画

彩色文本动画
// 颜色动画 gsap.to("#colorText", { color: "#ff6b6b", backgroundColor: "#4ecdc4", duration: 2, ease: "power2.inOut", yoyo: true, repeat: 1 }); // 文本动画 gsap.fromTo("#colorText", { fontSize: "24px" }, { fontSize: "48px", duration: 2, ease: "elastic.out" } );

6. SVG 动画

// SVG 属性动画 gsap.to("#svgCircle", { attr: { r: 80, fill: "#ff6b6b" }, duration: 2, ease: "bounce.out", yoyo: true, repeat: 1 }); // SVG 路径变形 gsap.to("#morphPath", { morphSVG: "M50,50 Q100,100 150,50 Q100,150 50,50", duration: 2, ease: "power2.inOut" });

7. 滚动触发器 (ScrollTrigger)

滚动我
// 滚动触发器 gsap.registerPlugin(ScrollTrigger); gsap.to("#scrollBox", { x: 200, rotation: 360, scrollTrigger: { trigger: "#scrollBox", start: "top 80%", end: "bottom 20%", scrub: true, markers: true } });

8. 高级功能

延迟
重复
回调
// 延迟动画 gsap.to("#advanced1", { y: -100, duration: 1, delay: 0.5, ease: "bounce.out" }); // 重复动画 gsap.to("#advanced2", { rotation: 360, duration: 1, repeat: 2, yoyo: true }); // 回调函数 gsap.to("#advanced3", { scale: 1.5, duration: 1, onStart: () => console.log("动画开始"), onComplete: () => { console.log("动画完成"); alert("动画完成！"); } });
